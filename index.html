🗄️ Estrutura de Tabelas Supabase

Crie as seguintes tabelas no seu projeto Supabase:

1. Tabela: eventos

CREATE TABLE eventos (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  nome TEXT NOT NULL,
  data DATE NOT NULL,
  local TEXT,
  categoria TEXT,
  descricao TEXT,
  destaque BOOLEAN DEFAULT false,
  imagem_url TEXT,
  created_at TIMESTAMP DEFAULT NOW()
);

2. Tabela: lotes

CREATE TABLE lotes (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  evento_id UUID REFERENCES eventos(id),
  nome TEXT NOT NULL,
  preco DECIMAL(10,2) NOT NULL,
  quantidade_total INTEGER NOT NULL,
  quantidade_vendida INTEGER DEFAULT 0,
  ativo BOOLEAN DEFAULT true,
  ordem INTEGER DEFAULT 0,
  created_at TIMESTAMP DEFAULT NOW()
);

3. Tabela: compradores

CREATE TABLE compradores (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  codigo_ingresso TEXT UNIQUE NOT NULL,
  nome TEXT NOT NULL,
  email TEXT NOT NULL,
  telefone TEXT,
  evento_id UUID REFERENCES eventos(id),
  lote_id UUID REFERENCES lotes(id),
  quantidade INTEGER NOT NULL,
  total DECIMAL(10,2) NOT NULL,
  cupom_usado TEXT,
  checkin BOOLEAN DEFAULT false,
  checkin_data TIMESTAMP,
  created_at TIMESTAMP DEFAULT NOW()
);

4. Tabela: cupons

CREATE TABLE cupons (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  codigo TEXT UNIQUE NOT NULL,
  tipo TEXT NOT NULL, -- 'percent' ou 'fixed'
  desconto DECIMAL(10,2) NOT NULL,
  usos INTEGER DEFAULT 0,
  limite INTEGER NOT NULL,
  ativo BOOLEAN DEFAULT true,
  created_at TIMESTAMP DEFAULT NOW()
);

5. Tabela: configuracoes

CREATE TABLE configuracoes (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  chave TEXT UNIQUE NOT NULL,
  valor JSONB,
  updated_at TIMESTAMP DEFAULT NOW()
);

🔐 Configuração RLS (Row Level Security)

Para maior segurança, habilite RLS nas tabelas:

-- Habilitar RLS em todas as tabelas
ALTER TABLE eventos ENABLE ROW LEVEL SECURITY;
ALTER TABLE lotes ENABLE ROW LEVEL SECURITY;
ALTER TABLE compradores ENABLE ROW LEVEL SECURITY;
ALTER TABLE cupons ENABLE ROW LEVEL SECURITY;
ALTER TABLE configuracoes ENABLE ROW LEVEL SECURITY;

-- Políticas para leitura pública (ajuste conforme necessário)
CREATE POLICY "Permitir leitura pública de eventos"
  ON eventos FOR SELECT
  USING (true);

CREATE POLICY "Permitir leitura pública de lotes"
  ON lotes FOR SELECT
  USING (true);

-- Políticas de escrita apenas para usuários autenticados
CREATE POLICY "Permitir inserção de compradores"
  ON compradores FOR INSERT
  WITH CHECK (true);
